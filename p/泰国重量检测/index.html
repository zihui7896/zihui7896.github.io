 public static byte[] HexStringToByteArray(string hex)
 {
     int length = hex.Length;
     if (length % 2 != 0)
     {
         throw new ArgumentException("十六进制字符串长度必须为偶数。");
     }
     byte[] bytes = new byte[length / 2];
     for (int i = 0; i < length; i += 2)
     {
         string hexPair = hex.Substring(i, 2);
         bytes[i / 2] = Convert.ToByte(hexPair, 16);
     }
     return bytes;
 }

 string _Data = string.Empty;
public void ReciveMsg(object sender, SerialDataReceivedEventArgs e)
 {
     SerialPort sp = (SerialPort)sender;

     // 读取字节数据
     int bytesToRead = sp.BytesToRead;
     byte[] buffer = new byte[bytesToRead];
     sp.Read(buffer, 0, bytesToRead);

     string hexData = BitConverter.ToString(buffer).Replace("-", "");
     //010304024E00009B9C
     
     _Data += hexData;
     MachineException.CallException(ExceptionType.NormalInfo, "SerialPort Received Bytes", _Data);
    
     // 检测结束符
     if (_Data.StartsWith("0103") && _Data.Length == 18)  // 根据需要变
     {
         MachineException.CallException(ExceptionType.NormalInfo, "_Data.EndsWith(\"0103\") && _Data.Length == 18", _Data);
         GlobalVariable.eReciveData = _Data;
         string show = _Data;
         _Data = string.Empty;
     }
   
 }

using NBVision.Shared;
using System;
using System.Collections.Generic;
using System.IO.Ports;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms;
using NBVision.UI.UserUI;
using Nobug2.Vision;
using Nobug;
namespace NBVision.Flow
{
    public class PortCommunication
    {
        public Action<string> laserMsgDisplay;
        public string Description;
        public SerialPort serialPort;
        public Action<string> myAction;
        public string Port { get; set; }

        public FormProductMessage formProduct;


        public List<string> listReciveDate = new List<string>();

        public string _hex_order = "AA";  // 通讯命令


        public bool isSignal = false;  // 默认无信号
        public PortCommunication(string description, string port = "COM5", int baudRate = 9600, Parity parity = Parity.Even, int dataBits = 8, StopBits stopBits = StopBits.One)
        {
            this.Description = description;
            this.Port = port;
            serialPort = new SerialPort();
            serialPort.PortName = port;
            serialPort.BaudRate = baudRate;
            serialPort.Parity = parity;
            serialPort.DataBits = dataBits;
            serialPort.StopBits = stopBits;
            serialPort.ReadTimeout = 2000;

            //serialPort.DataReceived += ReciveMsg; // 全局里面
        }
        public bool Open()
        {
            try
            {
                if (serialPort != null)
                {
                    if (!serialPort.IsOpen)
                    {
                        serialPort.Open();
                    }
                }
                return true;
            }
            catch (Exception ex)
            {
                //  MachineException.CallException(ExceptionType.NormalErr, "", ex.Message);
                MessageBox.Show("测试仪器" + ex.Message);
                return false;
            }
        }
        public void CloseComPort()
        {
            if (serialPort != null)
            {
                if (serialPort.IsOpen)
                {
                    serialPort.Close();
                }
            }
        }
    }
}
