public bool CodeJudge(string code, out string ErrMsg)
{
    try
    {
        if (code.Length != GlobalVariable.CurrentProgramConfig.PG110条码长度)
        {
            ErrMsg = "条码长度异常";
            return false;
        }
    re:
        if (!GlobalVariable.CurrentSystemConfig.PG110MES跳站检查)
        {
            if (!GlobalVariable.mysql.isOpen)
            {
                GlobalVariable.mysql = new MysqlHelper(GlobalVariable.CurrentSystemConfig.数据库名称, GlobalVariable.CurrentSystemConfig.数据库IP, "3306", "root", "root");
                string ErrMsg1 = string.Empty;
                bool rtn = GlobalVariable.mysql.Connect(out ErrMsg1);
                if (!rtn)
                {
                    MachineException.CallException(ExceptionType.NormalInfo, this._description, "数据库连接失败");
                    ErrMsg = "数据库连接失败";
                    return false;
                }
                else
                {
                    goto re;
                }
            }
            else
            {
                QueryHelper queryHelper = new QueryHelper();
                queryHelper.tableName = GlobalVariable.CurrentSystemConfig.PG110MES前站表名;

                queryHelper.L_SelectClounmName = new List<string>();
                queryHelper.L_SelectClounmName.Add("HK7");

                queryHelper.L_SelectItem = new List<SelectItem>();
                queryHelper.L_SelectItem.Add(new SelectItem("S/N", SelectRegion.等于, code));
                string msg = string.Empty;
                List<string> result = new List<string>();
                bool rtn = GlobalVariable.mysql.Query(queryHelper, out result, ref msg);
                if (!rtn)
                {
                    ErrMsg = "数据库查询异常";
                    return false;
                }

                if (result.Count == 0)
                {
                    ErrMsg = "前站未检测";
                    return false;
                }
                else
                {
                    if (result[result.Count - 1] == "false")
                    {
                        ErrMsg = "前站未烘烤";
                        return false;
                    }
                    else
                    {
                        ErrMsg = string.Empty;
                        return true;
                    }

                }
            }
        }
        else
        {
            ErrMsg = string.Empty;
            return true;
        }
    }
    catch (Exception ex)
    {
        ErrMsg = ex.ToString();
        return false;
    }
}

     public bool CheckRGB(HObject image,int rMin, int rMax, int gMin, int gMax, int bMin, int bMax)
     {


         HObject red = null, green = null, blue = null;
         HObject redRegion = null, greenRegion = null, blueRegion = null;
         HObject tempRegion = null, validRegion = null;

         try
         {

             // 2. 检查是否为三通道图像
             HTuple channels;
             HOperatorSet.CountChannels(image, out channels);
             if (channels.I != 3)
                 throw new Exception("错误：图像不是RGB三通道格式！");

             // 3. 分解RGB通道
             HOperatorSet.Decompose3(image, out red, out green, out blue);

             // 4. 对各通道进行阈值分割
             HOperatorSet.Threshold(red, out redRegion, rMin, rMax);
             HOperatorSet.Threshold(green, out greenRegion, gMin, gMax);
             HOperatorSet.Threshold(blue, out blueRegion, bMin, bMax);

             // 5. 计算三个通道的交集
             HOperatorSet.Intersection(redRegion, greenRegion, out tempRegion);
             HOperatorSet.Intersection(tempRegion, blueRegion, out validRegion);

             // 6. 计算有效区域像素数
             HTuple areaValid, row, col;
             HOperatorSet.AreaCenter(validRegion, out areaValid, out row, out col);

             // 7. 获取总像素数
             HTuple width, height;
             HOperatorSet.GetImageSize(image, out width, out height);
             long totalPixels = width * height;

             // 8. 判断结果
             if (areaValid.L == totalPixels)
             {
                 return true;
             }
             else
             {
                 double ratio = (double)areaValid.D / totalPixels * 100;
                 return false;
             }
         }
         catch (HalconException hex)
         {
             Console.WriteLine($"Halcon异常: {hex.Message}");
             return false;
         }
         finally
         {
             // 释放所有Halcon对象
             red.Dispose();
             green.Dispose();
             blue.Dispose();
             redRegion.Dispose();
             greenRegion.Dispose();
             blueRegion.Dispose();
             tempRegion.Dispose();
             validRegion.Dispose();
         }
     }
